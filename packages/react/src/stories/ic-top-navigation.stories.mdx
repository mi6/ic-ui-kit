  const [color, setColour] = useState("rgb(255, 201, 60)");

    setColour("rgb(35, 81, 142)");
  };

    setColour("rgb(255, 201, 60)");
  };

    <>

        Default theme
      </IcButton>
      <IcButton variant="primary" onClick={differentButtonClickHandler}>

    </>
  );
