export const defaultThemeArgs = {
  theme: "light",
  componentTheme: "light",
};

<Canvas>
  <Story
    name="Theme visualisation playground"
    args={defaultThemeArgs}
    parameters={{ loki: { skip: true }, layout: "fullscreen" }}
    argTypes={{
      theme: {
        options: ["system", "light", "dark"],
        control: {
          type: "inline-radio",
        },
      },
      componentTheme: {
        options: ["inherit", "light", "dark"],
        control: {
          type: "inline-radio",
        },
      },
    }}

      html`<div

          <div
            style="background-color: ${args.componentTheme == "dark"
              ? "black"

              ? "white"
              : null};width:250px"
          >
            <ic-typography theme=${args.componentTheme}
              >Theme test</ic-typography
            >
          </div>
        </ic-theme>
      </div>`
