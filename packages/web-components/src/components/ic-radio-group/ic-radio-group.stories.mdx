### Conditional dynamic

<Canvas>
  <Story name="Conditional dynamic">
    {(args) =>
      html`
        <script>
          function handleIcChange(ev) {

            .addEventListener("icChange", handleIcChange);
        </script>
        <ic-radio-group label="Conditional dynamic" name="1">
          <ic-radio-option
            additional-field-display="dynamic"
